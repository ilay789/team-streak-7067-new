async function init() { const errorBox = document.getElementById('error-box'); const selector = document.getElementById('year-selector'); try { // הוספת פרמטר למניעת שמירה במטמון (Cache) כדי שתמיד יטען את הקובץ הכי חדש const response = await fetch('seasons.xlsx?v=' + Math.random()); if (!response.ok) { throw new Error(`הקובץ seasons.xlsx לא נמצא בשרת.`); } const arrayBuffer = await response.arrayBuffer(); const data = new Uint8Array(arrayBuffer); const workbook = XLSX.read(data, { type: 'array' }); // קריאת הגיליון הראשון שבו יש נתונים const firstSheetName = workbook.SheetNames[0]; const worksheet = workbook.Sheets[firstSheetName]; // הפיכה ל-JSON const jsonData = XLSX.utils.sheet_to_json(worksheet); console.log("נתונים שנקראו:", jsonData); if (!jsonData || jsonData.length === 0) { throw new Error("קובץ האקסל נטען, אבל לא נמצאו בו שורות של נתונים. וודא שהכותרות בשורה הראשונה הן: Year, Challenge, Robot, Video."); } selector.innerHTML = ''; jsonData.sort((a, b) => b.Year - a.Year).forEach(item => { if(!item.Year) return; // דילוג על שורות ריקות const btn = document.createElement('button'); btn.innerText = item.Year; btn.className = 'year-btn'; btn.onclick = () => { document.querySelectorAll('.year-btn').forEach(b => b.classList.remove('active')); btn.classList.add('active'); document.getElementById('card-title').innerText = `עונת ${item.Year}`; document.getElementById('card-content').innerHTML = `

האתגר: ${item.Challenge || 'אין מידע'}

הרובוט: ${item.Robot || 'אין מידע'}

${item.Video ? `🎬 לצפייה בסרטון האתגר` : ''} `; }; selector.appendChild(btn); }); } catch (err) { document.getElementById('loading-text').style.display = 'none'; errorBox.style.display = 'block'; errorBox.innerHTML = `⚠️ בעיה בטעינה: ${err.message}`; console.error("Full Error:", err); } } window.onload = init;
